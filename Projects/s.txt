string splitString(string input, char split)
{
    string result = "";
    int startIndex = 0;

    for (int i = 0; i < input.length(); i++)
    {
        if (input[i] == split)
        {
            startIndex = i + 1;
            break;
        }
    }

    for (int i = startIndex; i < input.length(); i++)
    {
        result = result + input[i];
    }

    return result;
}
